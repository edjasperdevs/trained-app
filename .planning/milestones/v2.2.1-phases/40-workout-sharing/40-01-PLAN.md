**Camera integration:**
```typescript
import { Camera, CameraResultType, CameraSource } from '@capacitor/camera'

const capturePhoto = async (): Promise<string | undefined> => {
  try {
    const photo = await Camera.getPhoto({
      quality: 90,
      allowEditing: false,
      resultType: CameraResultType.DataUrl,
      source: CameraSource.Camera,
    })
    return photo.dataUrl
  } catch {
    return undefined // User cancelled or error
  }
}
```

**Props interface:**
```typescript
interface ShareBottomSheetProps {
  isOpen: boolean
  onClose: () => void
  onShare: (withPhoto: boolean, photoDataUrl?: string) => void
  isLoading?: boolean
}
```

**Behavior:**

- "Share without Photo" calls onShare(false)
