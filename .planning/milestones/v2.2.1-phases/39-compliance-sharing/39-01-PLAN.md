const isFullCompliance = data.workout && data.protein && data.meal && data.steps && data.sleep
// For recovery days (no workout scheduled), consider it full if all available items are checked

Note: Per feature brief, share button only shows on "full 5/5 compliance days". On recovery days (no workout), the user cannot achieve 5/5, so share button should NOT appear. Only show share when all 5 items are checked including workout.

4. Add milestone detection:
```tsx
function getMilestone(streak: number): string | undefined {
  if (streak === 7) return 'FIRST WEEK COMPLETE'
  if (streak === 30) return '30-DAY PROTOCOL'
  if (streak === 100) return '100 DAYS OF DISCIPLINE'
  return undefined
}
```

5. Add share handler:
```tsx
const handleShare = async () => {
  if (!cardRef.current || sharing) return
  setSharing(true)
  try {
    await shareComplianceCard(cardRef.current, obedienceStreak, totalDP, rankInfo.name)
  } finally {
    setSharing(false)
  }
}
```

6. In the submitted state section (after "Report Accepted." display), add share button ONLY when isFullCompliance is true:
```tsx
{submitted && isFullCompliance && (
  <Button
    onClick={handleShare}
    disabled={sharing}
    variant="outline"
    className="w-full mt-3 border-primary/30 text-primary font-heading uppercase tracking-widest"
  >
    {sharing ? 'Creating Card...' : 'Share Your Protocol'}
  </Button>
)}
```

Place this button AFTER the "Total Earned" display and BEFORE the "Continue" button.
