Add useEffect that runs on mount and when user returns to Home:
```typescript
useEffect(() => {
  const checkWeeklyReport = () => {
    const { shouldShowReport } = useWeeklyReportStore.getState()
    if (shouldShowReport()) {
      setShowWeeklyReport(true)
    }
  }

  // Check on mount
  checkWeeklyReport()

  // Also check when dpStore.lastActionDate changes (after a DP action)
  const unsubscribe = useDPStore.subscribe(
    (state) => state.lastActionDate,

  )

  return () => unsubscribe()
}, [])
```
