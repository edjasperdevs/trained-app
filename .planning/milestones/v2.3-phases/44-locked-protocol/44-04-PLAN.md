```typescript
import { useLockedStore } from '@/stores/lockedStore'
import { useRemindersStore } from '@/stores/remindersStore'

export async function scheduleLockedProtocolNotifications() {
  const lockedPrefs = useRemindersStore.getState().notificationPreferences.lockedProtocol
  const { activeProtocol, hasLoggedToday } = useLockedStore.getState()

  // Cancel any existing locked notifications first
  await cancelLockedProtocolNotifications()

  // Only schedule if enabled and protocol is active
  if (!lockedPrefs.enabled || !activeProtocol) return

  const protocolType = activeProtocol.protocolType

  if (protocolType === 'continuous') {
    // Schedule evening reminder if not logged today
    if (!hasLoggedToday) {
      await LocalNotifications.schedule({
        notifications: [{
          id: NOTIFICATION_IDS.LOCKED_REMINDER,
          title: 'Protocol reminder.',
          body: "You haven't logged today. Don't break the streak.",
          schedule: {
            at: getScheduleDate(lockedPrefs.time.hour, lockedPrefs.time.minute),
            allowWhileIdle: true,
          },
        }],
      })
    }
  } else if (protocolType === 'day_lock') {
    // Schedule morning "lock up" reminder
    const morningTime = lockedPrefs.time // For Day Lock, default should be 7am
    await LocalNotifications.schedule({
      notifications: [{
        id: NOTIFICATION_IDS.LOCKED_MORNING,
        title: 'Time to lock up.',
        body: 'Your protocol is waiting.',
        schedule: {
          at: getScheduleDate(morningTime.hour, morningTime.minute),
          allowWhileIdle: true,
        },
      }],
    })

    // Schedule optional evening soft reminder
    if (lockedPrefs.eveningReminder.enabled) {
      await LocalNotifications.schedule({
        notifications: [{
          id: NOTIFICATION_IDS.LOCKED_EVENING,
          title: 'End of day.',
          body: 'Protocol check.',
          schedule: {
            at: getScheduleDate(lockedPrefs.eveningReminder.time.hour, lockedPrefs.eveningReminder.time.minute),
            allowWhileIdle: true,
          },
        }],
      })
    }
  }
}

export async function cancelLockedProtocolNotifications() {
  try {
    await LocalNotifications.cancel({
      notifications: [
        { id: NOTIFICATION_IDS.LOCKED_REMINDER },
        { id: NOTIFICATION_IDS.LOCKED_MORNING },
        { id: NOTIFICATION_IDS.LOCKED_EVENING },
      ],
    })
  } catch { /* ignore if not found */ }
}
```
