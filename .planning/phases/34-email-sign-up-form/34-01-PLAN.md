Add the 4-segment password strength indicator and form validation logic.

1. **Password strength calculation:**
   ```typescript
   function getPasswordStrength(password: string): number {
     let strength = 0
     if (password.length >= 8) strength++
     if (/[A-Z]/.test(password)) strength++
     if (/[0-9]/.test(password)) strength++
     if (/[^A-Za-z0-9]/.test(password)) strength++
     return strength
   }
   ```

   Criteria (per auth brief):
   - 1 segment: 8+ characters
   - 2 segments: + uppercase letter
   - 3 segments: + number
   - 4 segments: + special character

2. **Password strength indicator component:**
   ```tsx
   function PasswordStrengthIndicator({ strength }: { strength: number }) {
     return (
       <div className="flex gap-1 mt-2">
         {[1, 2, 3, 4].map((segment) => (
           <div
             key={segment}
             className={`h-1 flex-1 rounded-full transition-colors ${
               segment <= strength ? 'bg-[#D4A853]' : 'bg-[#3A3A3A]'
             }`}
           />
         ))}
       </div>
     )
   }
   ```

   - 4 equal-width segments
   - Filled segments: gold #D4A853
   - Empty segments: #3A3A3A
   - h-1 rounded-full
   - Positioned below password field, above confirm password

   ```typescript
   const isValidEmail = (email: string) => {
     return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)
   ```

2. **Create account handler (per auth brief):**
   ```typescript
   const handleCreateAccount = async () => {
     if (!isFormValid || isLoading) return

     setIsLoading(true)
     setError(null)

     const { data, error } = await supabase.auth.signUp({
       email,
       password,
     })

     setIsLoading(false)

     if (error) {
       // Handle specific Supabase auth errors
       if (error.message.includes('already registered')) {
         setError('An account with this email already exists.')
       } else {
         setError(error.message)
       }
       return
     }

     // Session created - App.tsx routing will handle navigation to onboarding
   }
   ```
