   - Query 1: `assigned_workouts` for the client (already available from Plan 02's `fetchClientAssignments`)
   - Query 2: `workout_logs` where `assignment_id` IN (assignment IDs) and `completed = true`

   Alternatively, add a function to `useCoachTemplates` or a new helper in the Programs tab:
   ```typescript
   const fetchCompletedAssignments = async (clientId: string) => {
     // Get assigned workouts for this client
     const { data: assignments } = await client
       .from('assigned_workouts')
       .select('id, date, exercises, notes')
       .eq('client_id', clientId)
       .order('date', { ascending: false })
       .limit(20)

     // Get workout logs that reference these assignments
     const assignmentIds = assignments?.map(a => a.id) || []
     if (assignmentIds.length === 0) return []

     const { data: logs } = await client
       .from('workout_logs')
       .select('id, date, exercises, assignment_id, completed')
       .in('assignment_id', assignmentIds)
       .eq('completed', true)

     // Pair each log with its assignment
     return logs?.map(log => {
       const assignment = assignments?.find(a => a.id === log.assignment_id)
       return { log, assignment }
     }).filter(pair => pair.assignment) || []
   }
   ```
